\section{Current And Resistance}
\subsection{Electric Current}
$i=\frac{dq}{dt}$ 1ampere(1A)=1 coulomb per second(1C/s)\\
\subsection{Current Density}
$i=\int \:\vec{J}\cdot d\vec{A}$\\
$\vec{J}=\left(ne\right)\vec{v_e}$ : $\vec{v_e}$ drift speed\\
\subsection{Resistance and Resistivity}
$R=\frac{V}{i}$ : 1 ohm (1$\Omega $) = 1 volt per ampere = 1 V/A\\
$\rho =\frac{1}{\sigma }=\frac{\vec{E}}{\vec{J}}$ : $\rho$ Resistivity, $\sigma$ conductivity\\
$R=\rho \frac{L}{A}$\\
$p-p_0=p_0\alpha \left(T-T_0\right)$\\
\subsection{Power Potencia}
$P=iV$ : rate eletrical energy transfer\\
$P=i^2R=\frac{V^2}{R}$ : resistive dissipation\\
1 volt-ampere $1V\cdot A=\left(1\frac{J}{C}\right)\left(1\frac{C}{s}\right)=1\frac{J}{s}=1W$ 1 watt\\
$P=\frac{W}{\Delta t}$\\
